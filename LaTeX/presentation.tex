\section{Introduction}
\begin{frame}{Research Problem}
\begin{itemize}
\item Traditional RAG fails at \textbf{global sensemaking questions}
\item Example: "What are the main crime trends 2016-2023?"
\item RAG retrieves isolated chunks
\item GraphRAG provides \textbf{entity relationships} for broader context
\end{itemize}
\textbf{Goal}: Comparative analysis with controlled variables
\end{frame}

\begin{frame}{Controlled Setup}
\begin{block}{Same for Both Pipelines}
\begin{itemize}
\item LLM: Mistral 7B (via Ollama)
\item Dataset: 155 NCA PDFs → 1,529 chunks
\item Chunking: 500 words/chunk, 50 overlap
\end{itemize}
\end{block}
\textbf{Only difference}: Retrieval method (RAG vs GraphRAG)
\end{frame}

\section{Literature Review}
\begin{frame}{Related Work}
\begin{itemize}

\section{Methodology}
\begin{frame}{Traditional RAG Pipeline}
\begin{enumerate}
\item Query ChromaDB (all-MiniLM-L6-v2 embeddings)
\item Retrieve top-3 relevant chunks
\item Build prompt with context
\item Generate answer (Mistral 7B)
\end{enumerate}
\vspace{0.3cm}
\textbf{Code}: \texttt{rag\_pipeline.py}
\end{frame}

\begin{frame}{GraphRAG Pipeline - Graph Construction}
\textbf{Rule-Based Entity Extraction} (not LLM):
\begin{itemize}
\item Dates: \texttt{20XX} regex
\item Organizations: ALL-CAPS pattern
\item People: Title Case pattern
\item Crimes: Keyword matching
\end{itemize}
\vspace{0.3cm}
\textbf{Relationships}:
\begin{itemize}
\item Co-occurrence within 200 chars
\item Types: \texttt{co-occurs\_with}, \texttt{affiliated\_with}
\end{itemize}
\end{frame}

\begin{frame}{GraphRAG Pipeline - Enhanced Edges}
\textbf{SEMANTIC\_SIMILAR}:
\begin{itemize}
\item ChromaDB embeddings (cosine similarity > 0.75)
\item 1,169 edges added
\end{itemize}
\vspace{0.3cm}
\textbf{CROSS-CLUSTER\_BRIDGE}:
\begin{itemize}
\item K-Means cross-cluster connections
\item 500 edges added
\end{itemize}
\vspace{0.3cm}
\textbf{Louvain Communities}: 126 structural clusters
\end{frame}

\section{Discussion}
\begin{frame}{Key Findings}
\begin{itemize}
\item \textbf{GraphRAG excels at global understanding}:
\begin{itemize}
\item Multi-hop reasoning via graph traversal
\item Community structure for thematic summary
\item Entity-level context missing in RAG
\end{itemize}
\item \textbf{Rule-based extraction} > LLM-based:
\begin{itemize}
\item Deterministic, fast
\item No garbage names ("entity1", "entity2")
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Challenges \& Limitations}
\textbf{Challenges}:
\begin{itemize}
\item Ollama timeout: Increased to 120s
\item Metrics: LLM-as-judge failed, manual scoring needed
\end{itemize}
\vspace{0.3cm}
\textbf{Limitations}:
\begin{itemize}
\item Small dataset (155 PDFs)
\item Manual metrics (subjectivity)
\end{itemize}
\end{frame}

\section{Conclusion}
\begin{frame}{Conclusion}
\begin{itemize}
\item GraphRAG validates superiority for \textbf{global sensemaking}
\item Comparable response times (~7.2s)
\item Rule-based extraction: viable alternative to LLM-based
\item Higher quality, lower computational cost
\end{itemize}
\vspace{0.5cm}
\textbf{Future Work}:
\begin{itemize}
\item Expand dataset
\item Automated metrics scoring
\item Hierarchical community summaries
\end{itemize}
\end{frame}
